\\

\left(M_4+M_5\right)\ddot{z}_4=\frac{J_{1}\cdot \ddot{\theta}_1-C_m}{r_g}-\left(M_4+M_5\right)\cdot g\\
\Leftrightarrow
\\

\end{align*}

On trouve au final : 

\begin{align*}
