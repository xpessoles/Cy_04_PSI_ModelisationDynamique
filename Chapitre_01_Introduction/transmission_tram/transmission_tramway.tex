\subsubsection{Dimensionnement dynamique du système.}

\textbf{Données}
\begin{itemize}
\item Le référentiel associé au repère $R_0=\quadruplet{O_0}{\overrightarrow{x}_0}{\overrightarrow{y}_0}{\overrightarrow{z}_0}$ est supposé galiléen ;
\item Toutes les liaisons sont supposées parfaites ;
\item Masse du solide 4 : $M_4=67kg$ ;
\item Masse du solide 5 : $M5=6000kg$ ;
\item accélération de la pesanteur : $\overrightarrow{g}=-g\cdot \overrightarrow{z}_0$ avec $g=9,81m\cdot s^{-2}$ ;
\item Les solides 4 et 5 sont supposés encastrés ;
\item Moment d'inertie du solide 1 selon l'axe $\couple{O_1}{\overrightarrow{x}_0}$ : $J_{\couple{O_1}{\overrightarrow{x}_0}}(S_1)=J_1=10,5\times 10^{-3}kg\cdot m^2$.
\item Inerties des solides 2 et 3 négligeables ;
\item $C_m$ le couple qu'exerce le moteur sur le solide 1 (selon l'axe $\overrightarrow{x}_0$) ;
\item $C_{21}$ le moment de l'action de l'arbre 2 sur l'arbre 1 en projection sur \couple{O_1}{\overrightarrow{x}_0} : $\overrightarrow{M}_{O_1}(2\to 1)\cdot \overrightarrow{x}_0=C_{21}$ ;
\item $C_{23}$ le moment de l'action de l'arbre 2 sur l'arbre 3 en projection sur \couple{O_3}{\overrightarrow{z}_0} : $\overrightarrow{M}_{O_3}(2\to 3)\cdot \overrightarrow{x}_0=C_{23}$ ;
\item En supposant un rendement égal à 1 au niveau du réducteur on obtient la relation entre $C_{21}$ et $C_{23}$ : 
\begin{align*}
C_{23}=-\frac{p_{34}}{r_g\cdot 2\cdot \pi}\cdot C_{21}
\end{align*}
\item On rappelle le résultat de la partie précédente : 
\begin{align*}
\dot{z}_4=-r_g\cdot \dot{\theta}_1
\end{align*}
et quelle que soit la valeur trouvée pour $r_g$, la valeur utilisée dans la suite du sujet sera : $r_g=6,36\times 10^{-5}m$.
\end{itemize}
